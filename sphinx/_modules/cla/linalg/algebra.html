import numpy as np



[docs]def bilinear(mat, left=None, right=None):
    n, m = mat.shape
    assert n == m, "Matrix must be square"

    if left is None:
        left = np.ones(n)

    if right is None:
        right = np.ones(n)

    return left.T @ (mat @ right)




[docs]def sssolve(A, b, IN):
    OUT = ~IN
    n = A.shape[0]

    x = np.zeros(n)
    x[OUT] = b[OUT]

    bbb = b[IN] - A[IN, :][:, OUT] @ x[OUT]
    x[IN] = np.linalg.solve(A[IN, :][:, IN], bbb)

    np.isclose(A @ x - b, 0.0)

    return x

        self.covariance = covariance

[docs]    def update_weights(self, lamb):
        weights, _ = self._compute_weight(lamb)
        new_weights = np.copy(self.weights)
        new_weights[self.free] = weights
        return new_weights



[docs]    def free_asset_to_bound(self):
        A = self.covariance_free_inv
        xxx = np.sum(A, axis=1)
        s = np.sum(xxx)

        m = A @ self.mean_free

        C = -s * m + np.sum(m) * xxx
        lamb = -xxx / C
        return lamb
